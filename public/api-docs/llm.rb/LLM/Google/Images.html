Overview

The LLM::Google::Images class provides an images object for interacting with Google's Imagen text-to-image models via the Imagen API: https://ai.google.dev/gemini-api/docs/imagen

Examples:

#!/usr/bin/env ruby
require "llm"
llm = LLM.google(key: ENV["KEY"])
res = llm.images.create prompt: "A dog on a rocket to the moon"
IO.copy_stream res.images[0], "rocket.png"

def create(prompt:, n: 1, image_size: nil, aspect_ratio: nil, person_generation: nil, model: "imagen-4.0-generate-001", **params)
  req  = LLM::Transport::Request.post("/v1beta/models/#{model}:predict?key=#{key}", headers)
  body = LLM.json.dump({
    parameters: {
      sampleCount: n,
      imageSize: image_size,
      aspectRatio: aspect_ratio,
      personGeneration: person_generation
    }.compact.merge!(params),
    instances: [{prompt:}]
  })
  req.body = body
  res, span, tracer = execute(request: req, operation: "request")
  res = ResponseAdapter.adapt(res, type: :image)
  tracer.on_request_finish(operation: "request", model:, res:, span:)
  res
end
