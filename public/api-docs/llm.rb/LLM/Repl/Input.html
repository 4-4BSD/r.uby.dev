def on_char(window, char)
  if EOF.include?(char)
    :exit
  elsif BACKSPACE.include?(char)
    backspace
    :backspace
  elsif ENTER.include?(char)
    :submit
  elsif char == UP
    window.scroll_up
    :up
  elsif char == DOWN
    window.scroll_down
    :down
  elsif char == CTRL_A
    move_start
    :ctrl_a
  elsif char == CTRL_E
    move_end
    :ctrl_e
  elsif char == LEFT
    move_left
    :left
  elsif char == RIGHT
    move_right
    :right
  elsif String === char
    insert(char)
    :char
  else
    nil
  end
end

def lines(cols)
  sync_scroll(cols)
  text = to_s
  chunks = text.chars.each_slice(cols).map(&:join)
  chunks = [""] if chunks.empty?
  chunks[@scroll, height] || []
end

def cursor_pos(cols)
  sync_scroll(cols)
  [(cursor / cols) - @scroll, cursor % cols]
end
