def initialize(agent, options = {})
  @agent = agent
  @provider = agent.llm.name
  @buffer = +""
  @cursor = 0
  @scroll = 0
  @height = options.fetch(:height, 3)
  @last_char_at = nil
  @paste = false
end

def on_char(window, char, now)
  is_paste = lambda { @last_char_at and (now - @last_char_at) < PASTE_THRESHOLD }
  if CTRL[:D] == char
    delete
    :ctrl_d
  elsif CTRL[:A] == char
    move_start
    :ctrl_a
  elsif CTRL[:E] == char
    move_end
    :ctrl_e
  elsif CTRL[:F] == char
    move_forward
    :ctrl_f
  elsif CTRL[:Y] == char
    restore
    :ctrl_y
  elsif CTRL[:K] == char
    kill
    :ctrl_k
  elsif char == LEFT
    move_left
    :left
  elsif char == RIGHT
    move_right
    :right
  elsif BACKSPACE.include?(char)
    backspace
    :backspace
  elsif ENTER.include?(char)
    if @paste = is_paste.()
      insert("\n")
      :char
    else
      :submit
    end
  elsif char == UP
    window.scroll_up
    :up
  elsif char == DOWN
    window.scroll_down
    :down
  elsif String === char
    insert(char)
    :char
  else
    nil
  end
ensure
  @last_char_at = now if char
end

def cursor_pos
  scroll!
  before = to_s[0...cursor]
  line  = before.count("\n")
  col   = cursor - (before.rindex("\n") || -1) - 1
  [line - @scroll, col]
end

  @buffer[@cursor, @buffer.size] = ""
  @cursor = @buffer.size
end

def restore
  return unless @copy
  @buffer.insert(@cursor, @copy)
  @cursor += @copy.size
end
