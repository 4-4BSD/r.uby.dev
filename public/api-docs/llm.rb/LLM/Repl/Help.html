def call(name: nil)
  if name.nil?
    write("\n#{self.class.help}\n\n")
  elsif command = LLM::Command.find_by(name:)
    write("\n#{command.help}\n\n")
  else
    write "\nNo help for #{name} was found" \
          "\nThat command doesn't exist." \
          "\n\n"
  end
end
