Overview

Walks through an array from the last element backward, one step at a time. #next moves forward toward the end. Both methods clamp at the array boundaries.

def next
  if @items.empty?
    nil
  elsif @cursor >= @items.size - 1
    @items[@cursor]
  else
    @cursor += 1
    @items[@cursor]
  end
end

def prev
  if @items.empty?
    nil
  elsif @cursor <= 0
    @items[@cursor]
  else
    @cursor -= 1
    @items[@cursor]
  end
end
